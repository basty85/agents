# Python program to calculate the first 10,000 terms of the series

def calculate_series(n_terms):
    total = 0
    for i in range(n_terms):
        term = 1 / (2 * i + 1)  # generates odd numbers: 1, 3, 5, 7, ...
        if i % 2 == 0:
            total += term  # add for even index
        else:
            total -= term  # subtract for odd index
    return total * 4  # multiply the total by 4

# Calculate the series for the first 10,000 terms
result = calculate_series(10000)
print(result)
